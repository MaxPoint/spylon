from __future__ import absolute_import, print_function
from spylon.common import JVMHelpers


[docs]class SimpleJVMHelpers(JVMHelpers):

    def __init__(self, gateway):
        """

        Parameters
        ----------
        gateway : py4j.gateway.JavaGateway
            Generally constructed with `gateway = JavaGateway(GatewayClient(port=gateway_port), auto_convert=True)`
        """
        super(SimpleJVMHelpers, self).__init__()
        self._gateway = gateway

    @property
    def classloader(self):
        j = self.jvm.java.lang.Thread
        jthread = j.currentThread()
        cl = jthread.getContextClassLoader()
        return cl

    @property
    def jvm(self):
        return self.gateway.jvm

    @property
    def gateway(self):
        return self._gateway
